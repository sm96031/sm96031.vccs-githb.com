import pygame import random import time # Initialize Pygame pygame.init() # Define colors BLACK = (0, 0, 0) WHITE = (255, 255, 255) GREEN = (0, 255, 0) RED = (213, 50, 80) BLUE = (50, 153, 213) # Set the width and height of the screen WIDTH = 600 HEIGHT = 400 BLOCK_SIZE = 20 # Set the game clock clock = pygame.time.Clock() # Font settings font_style = pygame.font.SysFont("bahnschrift", 25) score_font = pygame.font.SysFont("comicsansms", 35) # Create the game screen screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption('Snake Game') # Function to display the current score def Your_score(score): value = score_font.render("Your Score: " + str(score), True, WHITE) screen.blit(value, [0, 0]) # Function to display messages def message(msg, color): mesg = font_style.render(msg, True, color) screen.blit(mesg, [WIDTH / 6, HEIGHT / 3]) # Function to draw the snake def draw_snake(block_size, snake_list): for x in snake_list: pygame.draw.rect(screen, GREEN, [x[0], x[1], block_size, block_size]) # Function to generate random food def generate_food(): return round(random.randrange(0, WIDTH - BLOCK_SIZE) / BLOCK_SIZE) * BLOCK_SIZE, round(random.randrange(0, HEIGHT - BLOCK_SIZE) / BLOCK_SIZE) * BLOCK_SIZE # Main game loop def gameLoop(): game_over = False game_close = False # Initial position of the snake x1 = WIDTH / 2 y1 = HEIGHT / 2 # Snake movement x1_change = 0 y1_change = 0 # Snake body snake_List = [] Length_of_snake = 1 # Generate the first food foodx, foody = generate_food() while not game_over: while game_close: screen.fill(BLUE) message("You Lost! Press Q-Quit or C-Play Again", RED) Your_score(Length_of_snake - 1) pygame.display.update() # Check for player input to restart or quit for event in pygame.event.get(): if event.type == pygame.KEYDOWN: if event.key == pygame.K_q: game_over = True game_close = False if event.key == pygame.K_c: gameLoop() # Event handling (checking for key presses) for event in pygame.event.get(): if event.type == pygame.QUIT: game_over = True if event.type == pygame.KEYDOWN: if event.key == pygame.K_LEFT: x1_change = -BLOCK_SIZE y1_change = 0 elif event.key == pygame.K_RIGHT: x1_change = BLOCK_SIZE y1_change = 0 elif event.key == pygame.K_UP: y1_change = -BLOCK_SIZE x1_change = 0 elif event.key == pygame.K_DOWN: y1_change = BLOCK_SIZE x1_change = 0 # Check if snake collides with the screen boundaries if x1 >= WIDTH or x1 < 0 or y1 >= HEIGHT or y1 < 0: game_close = True x1 += x1_change y1 += y1_change screen.fill(BLUE) # Draw the food pygame.draw.rect(screen, RED, [foodx, foody, BLOCK_SIZE, BLOCK_SIZE]) # Update snake's position snake_Head = [] snake_Head.append(x1) snake_Head.append(y1) snake_List.append(snake_Head) if len(snake_List) > Length_of_snake: del snake_List[0] # Check if snake collides with itself for x in snake_List[:-1]: if x == snake_Head: game_close = True draw_snake(BLOCK_SIZE, snake_List) Your_score(Length_of_snake - 1) pygame.display.update() # If the snake eats food, generate new food and increase the snake's length if x1 == foodx and y1 == foody: foodx, foody = generate_food() Length_of_snake += 1 # Set the speed of the game clock.tick(15) pygame.quit() quit() # Start the game loop gameLoop()
